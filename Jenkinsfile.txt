pipeline {
    agent any
    tools {
        sonarQubeScanner 'SonarScanner'  // Use the name configured in Global Tool Configuration
    }
    environment {
        SONARQUBE_URL = 'http://10.30.212.57:9000/'  // Update with your SonarQube server URL
    }
    stages {
        stage('Checkout SCM') {
            steps {
                checkout scm
            }
        }
        stage('SonarQube Analysis') {
            steps {
                script {
                    withSonarQubeEnv('SonarQube') {  // Ensure the name matches the one in your SonarQube configuration
                        sh 'sonar-scanner -Dsonar.projectKey=my_project -Dsonar.sources=src'
                    }
                }
            }
        }
    }
    post {
        failure {
            echo 'SonarQube analysis failed!'
        }
    }
}
